int main(){ 
    string s, t; cin >> s >> t;
    if(s==t){
        cout << 0 << endl;
        return 0;
    }
    int ss = s.size();
    int ts = t.size();
    rep(i, max(ss,ts)){
        if((ss<=i || ts<=i) || s[i]!=t[i]){
            cout << (i+1) << endl;
            return 0;
        }

}

## [C問題](https://atcoder.jp/contests/abc374/tasks/abc374_c)
### [提出コード](https://atcoder.jp/contests/abc374/submissions/58462285)

```c++
int main(){ 
    int n; cin >> n;
    vector<ll>k(n);
    rep(i,n){
        cin >> k[i];
    }
    ll ans = LINF;
    for (int i = 0; i < (1 << n); ++i){
        ll s1 = 0;
        ll s2 = 0;
        for (int j = 0; j < n; ++j){
            if (i & (1 << j)){
                s1+=k[j];        
            }else{
                s2+=k[j];
            }
        }
        ans = min(ans, max(s1, s2));
    }
    cout << ans << endl;
}
```
