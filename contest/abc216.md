## [B問題](https://atcoder.jp/contests/abc216/tasks/abc216_b)
### [提出コード](https://atcoder.jp/contests/abc216/submissions/25419036)
```c++
int main() {
  int n; cin >> n;
  vector<string>name(n);
  rep(i, n){
	  string s, t;
	  cin >> s >> t;
	  name[i] = s + "_" + t;
  }
 
	string ans = "No";
	for(int i = 0; i < n; ++i){
		for(int j = (i+1); j < n; ++j){
			if(name[i] == name[j]){
				ans = "Yes";
				break;
			}
		}
	}
 
    cout << ans << endl;
}
```
