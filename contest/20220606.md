```c++
int main(){
  int n; cin >> n;
  //桁　数字
  vector<vector<int>> dp(1000005, vector<int>(10, 0));
  for(int i=1;i<=9;++i){
	  dp[1][i] = 1;
  }
  for(int d=2;d<=n;++d){
	  for(int i=1;i<=9;++i){
		  for(int j=max(1,i-1);j<=min(9,i+1);++j){
			  dp[d][j]+=dp[d-1][i];
			  dp[d][j]%=MOD;
		  }
	  }
  }
  int ans = 0;
  for(int i=1;i<=9;++i){
    ans+=dp[n][i];
	ans%=MOD;
  }
  cout << ans << endl;
}
```
